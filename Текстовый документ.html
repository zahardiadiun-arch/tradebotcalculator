import tkinter as tk class Calculator(tk.Tk): def __init__(self): super().__init__() self.title("Калькулятор") self.geometry("300x400") self.configure(bg="#1e1e1e") self.resizable(False, False) self.expression = "" self.create_widgets() def create_widgets(self): # Поле ввода self.entry = tk.Entry( self, font=("Arial", 20), bd=5, relief="flat", justify="right", bg="#2d2d2d", fg="white" ) self.entry.pack(fill="x", padx=10, pady=10, ipady=10) # Кнопки btns = [ ("7", "8", "9", "/"), ("4", "5", "6", "*"), ("1", "2", "3", "-"), ("0", ".", "=", "+"), ] for row in btns: frame = tk.Frame(self, bg="#1e1e1e") frame.pack(expand=True, fill="both") for btn in row: button = tk.Button( frame, text=btn, font=("Arial", 18), fg="white", bg="#3c3c3c", activebackground="#505050", relief="flat", command=lambda b=btn: self.on_click(b) ) button.pack(side="left", expand=True, fill="both", padx=2, pady=2) # Кнопка очистки clear_btn = tk.Button( self, text="C", font=("Arial", 18), fg="white", bg="#d9534f", activebackground="#c9302c", relief="flat", command=self.clear ) clear_btn.pack(fill="x", padx=10, pady=5, ipady=5) def on_click(self, char): if char == "=": try: result = str(eval(self.expression)) self.entry.delete(0, tk.END) self.entry.insert(tk.END, result) self.expression = result except Exception: self.entry.delete(0, tk.END) self.entry.insert(tk.END, "Ошибка") self.expression = "" else: self.expression += str(char) self.entry.delete(0, tk.END) self.entry.insert(tk.END, self.expression) def clear(self): self.expression = "" self.entry.delete(0, tk.END) if __name__ == "__main__": app = Calculator() app.mainloop()
